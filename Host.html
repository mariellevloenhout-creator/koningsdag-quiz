<script>

const quiz=[

{q:"Hoe oud werd Willem-Alexander in 2026?",a:"57"},

{q:"Typisch Koningsdag eten?",a:"Tompouce"},

{q:"Kleur van de dag?",a:"Oranje"},

{q:"Kinderen op vrijmarkt doen?",a:"Spelen"},

{q:"Grootste chaos factor?",a:"Peuters"}

];


let i=-1;

let a=0,b=0;

let time=60;

let timer;


function next(){

clearInterval(timer);

i++;

if(i>=quiz.length){

document.getElementById("q").innerText="🎉 Einde quiz!";

return;

}

document.getElementById("q").innerText=quiz[i].q;

time=60;

document.getElementById("t").innerText=time;


timer=setInterval(()=>{

time--;

document.getElementById("t").innerText=time;

if(time<=0)clearInterval(timer);

},1000);

}


function show(){

alert("Antwoord: "+quiz[i].a);

}


function aPlus(){

a++;

document.getElementById("a").innerText=a;

}


function bPlus(){

b++;

document.getElementById("b").innerText=b;

}

</script>
